```python
import math

def gcd_of_strings(str1, str2):
    """
    –í—Ä–µ–º—è: O(n + m), –≥–¥–µ n = len(str1), m = len(str2)
    –ü–∞–º—è—Ç—å: O(n + m) –¥–ª—è –∫–æ–Ω–∫–∞—Ç–µ–Ω–∞—Ü–∏–∏ —Å—Ç—Ä–æ–∫
    """
    # –ü—Ä–æ–≤–µ—Ä–∫–∞: –º–æ–∂–Ω–æ –ª–∏ —Å–æ—Å—Ç–∞–≤–∏—Ç—å –∏–∑ –æ–¥–∏–Ω–∞–∫–æ–≤—ã—Ö –±–ª–æ–∫–æ–≤
    if str1 + str2 != str2 + str1:
        return ""

    # –î–ª–∏–Ω–∞ –æ–±—â–µ–≥–æ –¥–µ–ª–∏—Ç–µ–ª—è = –ù–û–î –¥–ª–∏–Ω
    gcd_len = math.gcd(len(str1), len(str2))

    # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –ø—Ä–µ—Ñ–∏–∫—Å –¥–ª–∏–Ω—ã gcd_len
    return str1[:gcd_len]


# –¢–µ—Å—Ç—ã
print(gcd_of_strings("ABCABC", "ABC"))      # "ABC"
print(gcd_of_strings("ABABAB", "ABAB"))     # "AB"
print(gcd_of_strings("LEET", "CODE"))       # ""
print(gcd_of_strings("ABABABAB", "ABAB"))   # "AB"
```

```python
def is_prime(num):
    if num < 2:
        return False
    for i in range(2, num):
        if num % i == 0:
            return False
    return True

    count = 0
    for i in range(2, n):

            count += 1
    return count
```

```python
def count_primes(n):
    """
    –†–µ—à–µ—Ç–æ –≠—Ä–∞—Ç–æ—Å—Ñ–µ–Ω–∞
    –í—Ä–µ–º—è: O(n log log n)
    –ü–∞–º—è—Ç—å: O(n)
    """
    if n < 2:
        return 0

    # –ò–∑–Ω–∞—á–∞–ª—å–Ω–æ –≤—Å–µ —á–∏—Å–ª–∞ —Å—á–∏—Ç–∞–µ–º –ø—Ä–æ—Å—Ç—ã–º–∏
    is_prime = [True] * n
    is_prime[0] = is_prime[1] = False  # 0 –∏ 1 –Ω–µ –ø—Ä–æ—Å—Ç—ã–µ

    # –í—ã—á—ë—Ä–∫–∏–≤–∞–µ–º –∫—Ä–∞—Ç–Ω—ã–µ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –ø—Ä–æ—Å—Ç–æ–≥–æ
    p = 2
    while p * p < n:
        if is_prime[p]:
            # –í—ã—á—ë—Ä–∫–∏–≤–∞–µ–º –∫—Ä–∞—Ç–Ω—ã–µ p, –Ω–∞—á–∏–Ω–∞—è —Å p¬≤
            for i in range(p * p, n, p):
                is_prime[i] = False
        p += 1

    # –ü–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ True
    return sum(is_prime)


# –¢–µ—Å—Ç—ã
print(count_primes(10))      # 4: [2, 3, 5, 7]
print(count_primes(100))     # 25
print(count_primes(1000))    # 168
```

```python
    if n < 3:
        return 0

    # –ú–∞—Å—Å–∏–≤ –¥–ª—è –Ω–µ—á—ë—Ç–Ω—ã—Ö —á–∏—Å–µ–ª (3, 5, 7, ...)
    # is_prime[i] —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç —á–∏—Å–ª—É 2*i + 3
    is_prime = [True] * (n // 2)

    p = 3
    while p * p < n:
        if is_prime[p // 2]:
            # –í—ã—á–µ—Ä–∫–∏–≤–∞–µ–º –∫—Ä–∞—Ç–Ω—ã–µ p (—Ç–æ–ª—å–∫–æ –Ω–µ—á—ë—Ç–Ω—ã–µ)
            for i in range(p * p, n, 2 * p):
                is_prime[i // 2] = False
        p += 2

    # +1 –¥–ª—è —á–∏—Å–ª–∞ 2
    return sum(is_prime) + 1

print(count_primes_optimized(10))  # 4
```

### –ó–∞–¥–∞—á–∞ 3: Power of Three (–°—Ç–µ–ø–µ–Ω—å —Ç—Ä–æ–π–∫–∏)

#### –ü–æ–¥—Ö–æ–¥ 1: –î–µ–ª–µ–Ω–∏–µ –≤ —Ü–∏–∫–ª–µ

**–ò–¥–µ—è:** –î–µ–ª–∏–º –Ω–∞ 3, –ø–æ–∫–∞ –≤–æ–∑–º–æ–∂–Ω–æ. –ï—Å–ª–∏ –≤ –∫–æ–Ω—Ü–µ –æ—Å—Ç–∞–ª–∞—Å—å 1 - —ç—Ç–æ —Å—Ç–µ–ø–µ–Ω—å —Ç—Ä–æ–π–∫–∏.

```python
def is_power_of_three_v1(n):
    """
    –í—Ä–µ–º—è: O(log‚ÇÉ n)
    –ü–∞–º—è—Ç—å: O(1)
    """
    if n < 1:
        return False

    while n % 3 == 0:
        n //= 3

    return n == 1


# –ü—Ä–∏–º–µ—Ä—ã
print(is_power_of_three_v1(27))   # True: 27 ‚Üí 9 ‚Üí 3 ‚Üí 1
print(is_power_of_three_v1(45))   # False: 45 ‚Üí 15 ‚Üí 5 (–æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º—Å—è)
```

**–ü–æ—à–∞–≥–æ–≤—ã–π —Ä–∞–∑–±–æ—Ä –¥–ª—è n = 27:**
```

–®–∞–≥ 2: 9 % 3 == 0 ‚úì, n = 9 // 3 = 3
–®–∞–≥ 3: 3 % 3 == 0 ‚úì, n = 3 // 3 = 1
–®–∞–≥ 4: n == 1 ‚úì, –≤–æ–∑–≤—Ä–∞—â–∞–µ–º True
```

```python
    """
    –ñ–∞–¥–Ω—ã–π –ø–æ–¥—Ö–æ–¥: –æ—Ç—Å–ª–µ–∂–∏–≤–∞–µ–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ –¥–æ—Å—Ç–∏–∂–∏–º—É—é –ø–æ–∑–∏—Ü–∏—é
    –í—Ä–µ–º—è: O(n)
    –ü–∞–º—è—Ç—å: O(1)
    """
    max_reach = 0  # –º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –¥–æ—Å—Ç–∏–∂–∏–º–∞—è –ø–æ–∑–∏—Ü–∏—è

    for i in range(len(nums)):
        # –ï—Å–ª–∏ —Ç–µ–∫—É—â–∞—è –ø–æ–∑–∏—Ü–∏—è –Ω–µ–¥–æ—Å—Ç–∏–∂–∏–º–∞
        if i > max_reach:
            return False

        # –û–±–Ω–æ–≤–ª—è–µ–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—É—é –¥–æ—Å—Ç–∏–∂–∏–º—É—é –ø–æ–∑–∏—Ü–∏—é
        max_reach = max(max_reach, i + nums[i])

        # –†–∞–Ω–Ω–∏–π –≤—ã—Ö–æ–¥: –µ—Å–ª–∏ —É–∂–µ –º–æ–∂–µ–º –¥–æ–π—Ç–∏ –¥–æ –∫–æ–Ω—Ü–∞

            return True

    return True
```

```python
    –í—Ä–µ–º—è: O(n)
```

```python
    total_cost = 0

        total_cost += cost[i]
        tank += gas[i] - cost[i]

    return start if total_gas >= total_cost else -1
```
